### 软件设计
实现功能：通过K3实现LED亮灭 外部中断0

	
	typedef unsigned int u8;
	typedef unsigned char u16;
	
	sbit KEY3=P3^2;
	sbit KEY4=P3^3;
	
	void delay_10us(u8 ten_us)
	{
		while(ten_us--);
	}
	
	sbit LED1=P2^0;
	
	
	void exti0_init(void)
	{
		IT0=1;
		EA=1;
		EX0=1;
	}
	
	
	
	void main()
	{
		exti0_init();
		while(1)
		{
			
		}
	}
		void exti0() interrupt 0
	{
			delay_10us(1000);
		  if(KEY3==0)
			{
				LED1=!LED1;
			}
	}
		void exti1() interrupt 2
	{
			delay_10us(1000);
		  if(KEY3==0)
			{
				LED1=!LED1;
			}
	}
