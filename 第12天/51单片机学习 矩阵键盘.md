# 矩阵按键介绍
#### 与独立按键区别
- 独立按键每个按键都要占用一个IO口，而矩阵按键可以节省IO口资源
- 一端是低电平的，易于编程/利用线翻转法和行列式扫描法检测
### 原理
- 行四个，列四个IO口占用
- 行列式扫描法
 一列输出0低电平，其他三列高电平然后检测该列，看第几行低电平就代表哪一个按下再返回出去（与独立按键扫描方法相同）
 - 线翻转法
1.使得行线全部为低，判断四列中哪个IO口读到低电平，则代表那一列中有按下
2.使得列线全为低，判断四行中哪个IO口读到低电平，则代表那一列中有按下
![[Pasted image 20260214192638.png]]
因为行全部都为低电平了，所以当哪个按键按下后，行与列导通低电平会传递到该列的IO口上

## 软件设计
- **实现功能**
#include <REGX52.H>
#include <intrins.h>

typedef unsigned int u16;
typedef unsigned char u8;

#define SMG P0
#define KEY_MATRIX_PORT P1

sbit LSA = P2^2;
sbit LSB = P2^3;
sbit LSC = P2^4;

u8 gsmg_code[17] = {
    0x3f, 0x06, 0x5b, 0x4f, 0x66, 0x6d, 0x7d, 0x07, 
    0x7f, 0x6f, 0x77, 0x7c, 0x39, 0x5e, 0x79, 0x71
};

void delay_10us(u16 ten_us)
{
    while(ten_us--);
}

u8 key_matrix_ranks_scan(void)
{
    u8 key_value = 0;

    // Scan Row 1 (S1, S2, S3, S4)
    KEY_MATRIX_PORT = 0x7f; 
    if(KEY_MATRIX_PORT != 0x7f)
    {
        delay_10us(1000);
        switch(KEY_MATRIX_PORT)
        {
            case 0x77: key_value = 1; break; // S4 -> Displays 0
            case 0x7b: key_value = 2; break; // S3 -> Displays 1
            case 0x7d: key_value = 3; break; // S2 -> Displays 2
            case 0x7e: key_value = 4; break; // S1 -> Displays 3
        }
        while(KEY_MATRIX_PORT != 0x7f);
    }

    // Scan Row 2 (S5, S6, S7, S8)
    KEY_MATRIX_PORT = 0xbf; 
    if(KEY_MATRIX_PORT != 0xbf)
    {
        delay_10us(1000);
        switch(KEY_MATRIX_PORT)
        {
            case 0xb7: key_value = 5; break; // S8 -> Displays 4
            case 0xbb: key_value = 6; break; // S7 -> Displays 5
            case 0xbd: key_value = 7; break; // S6 -> Displays 6
            case 0xbe: key_value = 8; break; // S5 -> Displays 7
        }
        while(KEY_MATRIX_PORT != 0xbf);
    }

    // Scan Row 3 (S9, S10, S11, S12)
    KEY_MATRIX_PORT = 0xdf; 
    if(KEY_MATRIX_PORT != 0xdf)
    {
        delay_10us(1000);
        switch(KEY_MATRIX_PORT)
        {
            case 0xd7: key_value = 9;  break; // S12 -> Displays 8
            case 0xdb: key_value = 10; break; // S11 -> Displays 9
            case 0xdd: key_value = 11; break; // S10 -> Displays A
            case 0xde: key_value = 12; break; // S9  -> Displays B
        }
        while(KEY_MATRIX_PORT != 0xdf);
    }

    // Scan Row 4 (S13, S14, S15, S16)
    KEY_MATRIX_PORT = 0xef; 
    if(KEY_MATRIX_PORT != 0xef)
    {
        delay_10us(1000);
        switch(KEY_MATRIX_PORT)
        {
            case 0xe7: key_value = 13; break; // S16 -> Displays C
            case 0xeb: key_value = 14; break; // S15 -> Displays D
            case 0xed: key_value = 15; break; // S14 -> Displays E
            case 0xee: key_value = 16; break; // S13 -> Displays F
        }
        while(KEY_MATRIX_PORT != 0xef);
    }

    return key_value;
}

void main()
{
    u8 key = 0;
    LSC = 0; LSB = 0; LSA = 0; 

    while(1)
    {
        key = key_matrix_ranks_scan(); 
        if(key != 0) 
        {
            SMG = gsmg_code[key - 1]; 
        }
    }
}
